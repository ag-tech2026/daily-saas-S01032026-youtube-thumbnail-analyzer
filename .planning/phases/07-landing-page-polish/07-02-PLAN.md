**Step 6: Create src/app/global-error.tsx**
React error boundary for unhandled render errors. Must be "use client":
```typescript
"use client";

import * as Sentry from "@sentry/nextjs";
import { useEffect } from "react";

export default function GlobalError({
  error,
  reset,
}: {
  error: Error & { digest?: string };
  reset: () => void;
}) {
  useEffect(() => {
    Sentry.captureException(error);
  }, [error]);

  return (
    <html>
      <body>
        <div style={{ padding: "2rem", textAlign: "center" }}>
          <h2>Something went wrong!</h2>
          <button onClick={() => reset()} style={{ marginTop: "1rem", padding: "0.5rem 1rem" }}>
            Try again
          </button>
        </div>
      </body>
    </html>
  );
}
```

import { withSentryConfig } from "@sentry/nextjs";

Do NOT include `org` or `project` in the config — those are set via SENTRY_ORG and SENTRY_PROJECT env vars, or during CI. Keeping them out of code avoids hardcoding project-specific values.
